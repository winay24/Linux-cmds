								[DIRECTORY RELATED COMMANDS]

ls 			--> 	list of parent directory
ls -a			-->	-a gives hidden files(ex : .git)
ls -l			-->	-l gives all properties such as (read, write permisssion and root)
ls -R			-->	-R gives list files in sub directories

-----------------------------------------------------------------------------------
pwd 			--> 	present working directory
clear			-->	clear prompt

-----------------------------------------------------------------------------------
cd			-->	change to root directory
cd [path]		-->	change to given path directory
cd ..			-->	change to parent directory from current directory
cd -			-->	change to previous directory from current
cd ~[user]		-->	change to different user

-----------------------------------------------------------------------------------
mkdir [name]	-->	creates empty directory
mkdir [path/name]	-->	creates path at destination path
mkdir -v fo1 fo2	-->	creates and prints msg once folder created

-----------------------------------------------------------------------------------
rmdir [name]	-->	removes empty directory
rmdir [path/name]	-->	rmoves particular directory in the given path
rmdir f1 f2 f3	-->	removes multiple directories at atime
